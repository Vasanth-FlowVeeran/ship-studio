```tsx
const handleSavePort = async (newPort: number) => {
  if (!currentProject) return;
  try {
    // 1. Persist to project.json
    await invoke('set_dev_server_port', { projectPath: currentProject.path, port: newPort });
    // 2. Update React state
    setDevServerPort(newPort);
    // 3. Close modal
    closeProjectSettings();
    // 4. Restart dev server with new port (use portOverride to avoid stale closure)
    await restartDevServer(currentProject.path, newPort);
    showToast('Port updated and server restarted', 'success');
  } catch (error) {
    showToast('Failed to save port setting', 'error');
  }
};
```

Note: `restartDevServer` is the renamed `handleRestartDevServer` from `useDevServer` (aliased in App.tsx line ~119 as `handleRestartDevServer: restartDevServer`). It now accepts the optional `portOverride` parameter.
